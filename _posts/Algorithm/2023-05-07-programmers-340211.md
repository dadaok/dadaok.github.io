import java.util.*;

public class Solution {
    public int solution(int[][] points, int[][] routes) {
        Map<Integer, int[]> pointMap = new HashMap<>();
        for (int i = 0; i < points.length; i++) {
            pointMap.put(i + 1, points[i]);
        }

        Map<String, Integer> positionTimeMap = new HashMap<>();
        int collisionCount = 0;

        for (int[] route : routes) {
            int time = 0;
            int[] current = pointMap.get(route[0]);

            // 시작 위치 기록
            String key = time + "," + current[0] + "," + current[1];
            positionTimeMap.put(key, positionTimeMap.getOrDefault(key, 0) + 1);

            for (int i = 1; i < route.length; i++) {
                int[] next = pointMap.get(route[i]);
                int r = current[0];
                int c = current[1];

                // r 좌표 이동
                while (r != next[0]) {
                    r += (next[0] > r) ? 1 : -1;
                    time++;
                    key = time + "," + r + "," + c;
                    positionTimeMap.put(key, positionTimeMap.getOrDefault(key, 0) + 1);
                }

                // c 좌표 이동
                while (c != next[1]) {
                    c += (next[1] > c) ? 1 : -1;
                    time++;
                    key = time + "," + r + "," + c;
                    positionTimeMap.put(key, positionTimeMap.getOrDefault(key, 0) + 1);
                }

                current = next;
            }
        }

        for (int count : positionTimeMap.values()) {
            if (count > 1) {
                collisionCount++;
            }
        }

        return collisionCount;
    }
}
